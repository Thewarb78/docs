mkdir -p ~/tools
cd ~/tools

# Get Spark 3.5.3
curl -fLO https://archive.apache.org/dist/spark/spark-3.5.3/spark-3.5.3-bin-hadoop3.tgz

# Extract
tar -xzf spark-3.5.3-bin-hadoop3.tgz

# Set env for this shell
export SPARK_HOME=~/tools/spark-3.5.3-bin-hadoop3
export PATH="$SPARK_HOME/bin:$PATH"

# Check
spark-submit --version
kubectl get pods -n spark

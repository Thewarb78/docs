# kubeconfigs folder relative to this script
$configDir = ".\kubeconfigs"

if (-not (Test-Path $configDir)) {
    Write-Error "kubeconfigs folder not found: $configDir"
    exit
}

# Find kubeconfig files
$configs = Get-ChildItem $configDir -File |
    Where-Object { $_.Extension -in ".yaml", ".yml", ".config" }

if (-not $configs) {
    Write-Error "No kubeconfig files found."
    exit
}

# Show selection window
$selection = $configs |
    Select-Object @{
        Name = "Cluster"
        Expression = { $_.BaseName }
    }, FullName |
    Out-GridView -Title "Select Kubernetes Cluster" -PassThru

if (-not $selection) {
    Write-Host "No cluster selected."
    exit
}

# Set kubeconfig for current shell session
$env:KUBECONFIG = $selection.FullName

Write-Host ""
Write-Host "Using cluster: $($selection.Cluster)"
Write-Host "KUBECONFIG=$env:KUBECONFIG"
Write-Host ""

kubectl config current-context

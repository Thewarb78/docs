apiVersion: apps/v1
kind: Deployment
metadata:
  name: spark-history-server
  namespace: spark
spec:
  replicas: 1
  selector:
    matchLabels:
      app: spark-history-server
  template:
    metadata:
      labels:
        app: spark-history-server
    spec:
      serviceAccountName: spark
      imagePullSecrets:
        - name: -pull-secret
      containers:
        - name: history-server
          image: <your-arti-image>
          command: ["/opt/spark/bin/spark-class"]
          args: ["org.apache.spark.deploy.history.HistoryServer"]
          ports:
            - containerPort: 18080
          env:
            - name: HADOOP_CONF_DIR
              value: /etc/hadoop/conf
            - name: SPARK_HISTORY_OPTS
              value: "-Dspark.history.fs.logDirectory=s3a://<bucket>/spark-history/"
          volumeMounts:
            - name: core-site
              mountPath: /etc/hadoop/conf
              readOnly: true
            - name: tmp
              mountPath: /tmp
      volumes:
        - name: core-site
          secret:
            secretName: spark-core-site
        - name: tmp
          emptyDir: {}

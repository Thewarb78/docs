  --issuer "https://" `
  --client-id "pinniped-cli" `
  --scopes "offline_access,openid,pinniped:request-audience,username,groups" `
  --ca-bundle "C:\k8s\ca-bundle.pem" `
  --request-audience "cluster" `
  --upstream-identity-provider-name "eldap" `

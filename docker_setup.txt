from pyspark.sql import SparkSession
from pyspark.sql.functions import col, rand, sum as spark_sum, count

spark = (
    SparkSession.builder
    .appName("spark-k8s-stress-test")
    .getOrCreate()
)

sc = spark.sparkContext
sc.setLogLevel("INFO")

num_rows = 50_000_000
num_partitions = 200

print(f"Creating {num_rows} rows across {num_partitions} partitions")

df = (
    spark.range(0, num_rows, 1, numPartitions=num_partitions)
    .withColumn("key", (col("id") % 10000))
    .withColumn("value", rand())
)

print("Running shuffle-heavy aggregation")

result = (
    df.repartition(num_partitions, "key")
    .groupBy("key")
    .agg(
        count("*").alias("row_count"),
        spark_sum("value").alias("value_sum")
    )
)

print("Forcing execution")
result.count()

print("Running sort")
result.orderBy(col("value_sum").desc()).count()

print("Done")

spark.stop()

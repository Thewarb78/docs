  --build-arg BASE_IMAGE=${BASE_IMAGE} \
  --build-arg SPARK_VERSION=${SPARK_VERSION} \
  --build-arg PYTHON_VERSION=${PYTHON_VERSION} \

./build.sh versions/spark-3.5.3-py311.env

IMAGE_NAME=artifactory.company.com/spark/spark-base
IMAGE_TAG=3.5.3-py3.11-u22-r1

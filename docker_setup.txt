function Use-KubeCluster {

    # Folder relative to this script
    $configDir = Join-Path $PSScriptRoot "kubeconfigs"

    if (-not (Test-Path $configDir)) {
        Write-Error "kubeconfigs folder not found: $configDir"
        return
    }

    # Find kubeconfig files
    $configs = Get-ChildItem -Path $configDir -File |
        Where-Object { $_.Extension -in ".yaml", ".yml", ".config" }

    if (-not $configs) {
        Write-Error "No kubeconfig files found in $configDir"
        return
    }

    # Build selection list
    $selection = $configs |
        Select-Object @{
            Name = "Cluster"
            Expression = { $_.BaseName }
        }, FullName |
        Out-GridView -Title "Select Kubernetes Cluster" -PassThru

    if (-not $selection) {
        Write-Host "No cluster selected."
        return
    }

    # Set for current session only
    $env:KUBECONFIG = $selection.FullName

    Write-Host ""
    Write-Host "Active cluster: $($selection.Cluster)"
    Write-Host "KUBECONFIG:"
    Write-Host "  $env:KUBECONFIG"
    Write-Host ""

    kubectl config current-context
}
